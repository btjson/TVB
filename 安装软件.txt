1. 安装软件
# 根据你的服务器架构选择，下面以 Linux AMD64 为例
sudo apt update
sudo apt install ffmpeg

mkdir stream
cd stream
wget https://github.com/nilaoda/N_m3u8DL-RE/releases/download/v0.2.1-beta/N_m3u8DL-RE_Beta_linux-x64_20240828.tar.gz
tar -zxvf N_m3u8DL-RE_Beta_linux-x64_20240828.tar.gz
mv N_m3u8DL-RE_Beta_linux-x64/N_m3u8DL-RE ./m3u8dl

wget https://github.com/shaka-project/shaka-packager/releases/download/v3.2.0/packager-linux-x64



chmod +x m3u8dl packager-linux-x64
